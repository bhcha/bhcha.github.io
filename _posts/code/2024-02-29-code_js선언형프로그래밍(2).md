### 2. filter
```
[1,2,3] -> f(x) { return x%2;} -> [2] 
```
```javascript
const filter = (f, iter) => {
    let res = [];
    for (const a of iter) {
        if (f(a)) res.push(a);
    }
    return res;
};
```

```javascript
const reduce = (f, acc, iter) => {
    if (!iter) {
        iter = acc[Symbol.iterator]();
        acc = iter.next().value;
    }
    for (const a of iter) {
        acc = f(acc, a);
    }
    return acc;
};

const add = (acc, curr) => acc + curr;
```

```javascript
const take = (l, iter) => {
```
